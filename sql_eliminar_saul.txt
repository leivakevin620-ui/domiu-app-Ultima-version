-- SQL PARA ELIMINAR CUENTAS DUPLICADAS DE SAÚL
-- Ejecutar en Supabase → SQL Editor

-- 1. Ver cuentas duplicadas
SELECT id, email, nombre, created_at 
FROM profiles 
WHERE LOWER(nombre) LIKE '%saúl%' OR LOWER(nombre) LIKE '%saul%'
ORDER BY created_at DESC;

-- 2. Ver repartidores duplicados
SELECT id, user_id, nombre, telefono, created_at
FROM repartidores
WHERE LOWER(nombre) LIKE '%saúl%' OR LOWER(nombre) LIKE '%saul%'
ORDER BY created_at DESC;
